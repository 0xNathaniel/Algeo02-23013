def add_latihan(list_latihan, id_latihan, tipe_latihan, intensitas_latihan, jadwal_latihan):
    # KAMUS LOKAL
        # tuple_latihan : tuple
        # list_latihan : list of tuple
        # id_latihan : integer
        # tipe_latihan : string
        # intensitas_latihan : integer
        # jadwal_latihan : datetime
    # ALGORITMA
    tuple_latihan = (id_latihan, tipe_latihan, intensitas_latihan, jadwal_latihan)
    list_latihan.append(tuple_latihan)

def delete_latihan(list_latihan, id_latihan):
    # KAMUS LOKAL
        # list_latihan : list of tuple
        # id_latihan : integer
        # index : integer
        # found : boolean
    # ALGORITMA
    index = 0
    found = False
    while (not found) and (index < len(list_latihan)):
        if list_latihan[index][0] == id_latihan:
            found = True
        else:
            index += 1
    list_latihan.pop(index)

def edit_latihan(list_latihan, id_latihan, tipe_latihan, intensitas_latihan, jadwal_latihan):
    # KAMUS LOKAL
        # list_latihan : list of tuple
        # id_latihan : integer
        # tipe_latihan : string
        # intensitas_latihan : integer
        # jadwal_latihan : datetime
        # index : integer
        # found : boolean
    # ALGORITMA
    index = 0
    found = False
    while (not found) and (index < len(list_latihan)):
        if list_latihan[index][0] == id_latihan:
            found = True
        else:
            index += 1
    list_latihan[index] = (id_latihan, tipe_latihan, intensitas_latihan, jadwal_latihan)
